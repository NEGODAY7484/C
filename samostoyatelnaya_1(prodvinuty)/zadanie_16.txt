#include <iostream>
#include <cmath>

using namespace std;

int main() {
    double V1, V2, P1, P2; // входные данные: объем 1, объем 2, давление 1, давление 2
    double V; // объем сыпучего вещества
    
    cout << "Введите объем 1 (м^3): ";
    cin >> V1;
    cout << "Введите давление 1 (Па): ";
    cin >> P1;
    cout << "Введите объем 2 (м^3): ";
    cin >> V2;
    cout << "Введите давление 2 (Па): ";
    cin >> P2;
    
    V = (P1 * V1 + P2 * V2) / (P1 + P2);

    V = (V < 0) ? 0 : V; // проверка на отрицательность
    
    cout << "Объем сыпучего вещества: " << V << " м^3" << endl;
    
    return 0;
}

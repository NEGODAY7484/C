#include <iostream>
using namespace std;

int main() {
    const int N = 7;   // число строк
    const int M = 3;   // число столбцов
    int arr[N][M];
    int count[M] = {0};  // инициализируем нулями
                         // для подсчета количества элементов в каждом столбце

    // заполнение массива случайными значениями
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            arr[i][j] = rand() % 10;
            cout << arr[i][j] << " ";
            if (arr[i][j] == 7) {
                count[j]++;   // если элемент равен 7, увеличиваем счетчик для соответствующего столбца
            }
        }
        cout << endl;
    }

    // вывод количества элементов для каждого столбца
    for (int j = 0; j < M; j++) {
        cout << "Количество элементов равных 7 в столбце " << j+1 << ": " << count[j] << endl;
    }

    return 0;
}
